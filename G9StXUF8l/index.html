0x02 脚本实现:

生成字典:

def main():
    
    fuzz_zs = ['/*', '*/', '/*!', '*', '=', '`', '!', '@', '%', '.', '-', '+', '|', '%00']
    fuzz_sz = ['', ' ']
    fuzz_ch = ["%0a", "%0b", "%0c", "%0d", "%0e", "%0f", "%0g", "%0h", "%0i", "%0j" ,"%0k", "%0m", "%0n"]
    fuzz = (fuzz_ch + fuzz_sz + fuzz_zs)
    for a in fuzz:
        for b in fuzz:
            for c in fuzz:
                for d in fuzz:
                    for e in fuzz:
                        for f in fuzz:
                            st = a+b+c+d+e+f
                            print("[+]"+st)
                            with open(r'SQLWaF_Baypass_Dict.txt','a+') as f:
                                f.write(st+'\n')
                        
if __name__ == "__main__":
    main()
